28def draw_Lexis_Diagram_Real(data,language = "English",isDark = False): 

29 # Adding a new column to the DataFrame 

30 data['int_c'] = data['v'] + data['t'] 

31 # Grouping the data by 't' and 'int_c' columns 

32 grouped_data = data.groupby(['t','int_c']).agg(def_num = ('y','sum'), 

33 number = ('t','count'),dr=('y','sum')) 

34 # Calculating the weighted mean 

35 grouped_data["dr"] = grouped_data["dr"]/grouped_data["number"] 

36 

37 

38 #generate index for x,y,z 

39 int_c_indices = [] 

40 t_indices = [] 

41 for index in grouped_data.index: 

42 int_c_indices.append(index[1]) 

43 t_indices.append(index[0]) 

44 int_c_indices = list(set(int_c_indices)) 

45 t_indices = list(set(t_indices)) 

46 

47 # Creating a list of all possible index combinations 

48 all_index = [] 

49 for y in t_indices: 

50 for x in int_c_indices: 

51 all_index.append((y, x)) 

52 

53 # Adding None for missing data 

54 z_values = [] 

55 for index in all_index: 

56 if index in grouped_data.index: 

57 z_values.append(grouped_data._get_value(index, 'dr')) 

58 else: 

59 z_values.append(None) 

60 

61 # Reshaping the list into a matrix 

62 default_rate = [] 

63 for line in np.reshape(z_values, (len(t_indices), len(int_c_indices))): 

64 default_rate.append(list(line)) 

65 

66 # Creating a list of time intervals 

67 c_indices =[] 

68 for i in int_c_indices: 

69 c = start_date + relativedelta(months=i) 

70 c_indices.append(c) 

71 c_indices 

72 

73 # Defining a list of color themes 

74 theme = ["hot_r","YlGnBu","OrRd","greys",[[0, 'rgb(255,255,204)'], 

75 [1, 'rgb(97,97,189)']], [[0, 'rgb(205,255,153)'], 

76 [1, 'rgb(51,102,0)']],[[0, 'rgb(153,255,255)'],

78 

79 theme_name = ["hot_r","YlGnBu","OrRd","greys", 

80 "yellows","greens","blues","Spectral","PiYG"] 

81 default_rate = np.array(default_rate).astype(float) 

82 # Visualizing the Lexis diagram for each theme 

83 for i in range(0,len(theme)): 

84 visualize_lexis_diagram(theme[i],theme_name[i], 

85 int_c_indices,t_indices,default_rate, 

86 "",language,isDark) 

87 visualize_lexis_diagram(theme[i],theme_name[i],

89 "sinh_",language,isDark) 

90 visualize_lexis_diagram(theme[i],theme_name[i],
