98""" 

99This function calculates the effect of variables on the response variable and returns relevant information. 

100Parameters: 

101data (pandas DataFrame): The input data for Age-Period-Cohort analysis. 

102""" 

103def get_effect(data): 

104 

105 # Add a column to the DataFrame that combines 'v' and 't' columns 

106 data['c'] = data['v'] + data['t'] 

107 # Group the data by 't', 'v', and 'c' columns, then aggregate data  

108 # based on count, sum of 'y', and sum of 'pd' 

109 data = data.groupby(['t','v','c']).agg(number = ('y','count'), 

110 dr=('y','sum'),pd=('pd','sum')) 

111 

112 # Compute the average of 'dr' and 'pd' for each group based on 'number' 

113 data["dr"] = data["dr"]/data["number"] 

114 data['pd'] = data['pd']/data['number'] 

115 

116 # Extract 't', 'v', and 'c' from the index of 'data' 

117 t_list,v_list,c_list = [],[],[] 

118 for (i,j,k) in data.index: 

119 t_list.append(i) 

120 v_list.append(j) 

121 c_list.append(k) 

122 

123 

124 data['t'] = t_list 

125 data['v'] = v_list 

126 data['c'] = c_list 

127 

128 # Prepare 'yTrain1', 'yTrain2', and 'train_encode' based on 'data' 

129 yTrain1 = data['pd'] 

130 yTrain2 = data['dr'] 

131 

132 # One-hot encode age, cohort, and 'c' values 

133 train_encode_simplify = data[['v','t','c']] 

134 train_encode = pd.get_dummies(train_encode_simplify, 

135 columns = ['v','t','c']) 

136 

137 # Create feature variable arrays for Ridge regression 

138 xTrain_encode = train_encode.values 

139 

140 # Fit linear regression models for pd and dr 

141 effect_list_predicted = Ridge(alpha=0.001) 

142 effect_list_predicted.fit(xTrain_encode,yTrain1) 

143 

144 effect_list_real = Ridge(alpha=2) 

145 effect_list_real.fit(xTrain_encode,yTrain2) 

146 

147 # Create arrays of unique age, cohort, and 'c' values for creating Lexis diagrams 

148 v = data['v'].unique() 

149 len_v = len(v) 

150 t = data['t'].unique() 

151 len_t = len(t) 

152 c = data['c'].unique() 

153 len_c = len(c) 

154 

155 # Create a 2D array of predicted pd values to create a Lexis diagram 

156 default_rate = get_predicted_default_rate(train_encode,t,c,effect_list_predicted) 

157 

158 # Create a 2D array of predicted dr values to create a Lexis diagram 

159 default_rate_real = get_predicted_default_rate(train_encode,t,c,effect_list_real) 

160 

161 

162 

163 return { 'len_t': len_t, "len_v": len_v, "len_c":len_c,"effect_list_predicted": 

164 effect_list_predicted,"effect_list_real":effect_list_real, 

165 'v':v, 't':t, 'c':c, "default_rate":default_rate, 

166 "default_rate_real":default_rate_real}

169""" 

170Create a 2D array of predicted default rate values to create a Lexis diagram 

171Parameter:  

172train_encode: encoded a,p,c value 

173t: age list 

174c: period list 

175effect_list_model: model to predict default rate 

176""" 

177def get_predicted_default_rate(train_encode,t,c,effect_list_model): 

178 # Create a 2D array of predicted pd values to create a Lexis diagram 

179 default_rate = [] 

180 for i in t: 

181 templist = [] 

182 for j in c: 

183 try: 

184 encode = train_encode.loc[i].loc[j-i] 

185 y = effect_list_model.predict(encode.values)[0] 

186 except: 

187 y = np.nan 

188 templist.append(y) 

189 default_rate.append(templist) 

190 return default_rate
