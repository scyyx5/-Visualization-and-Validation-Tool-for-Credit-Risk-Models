224def visit_tree(node, previsit, postvisit): 

225 # type: (Module, Callable[[AstNode, Optional[Token]], Tuple[Optional[Token], Optional[Token]]], Optional[Callable[[AstNode, Optional[Token], Optional[Token]], None]]) -> None 

226 """ 

227 Scans the tree under the node depth-first using an explicit stack. It avoids implicit recursion 

228 via the function call stack to avoid hitting 'maximum recursion depth exceeded' error. 

229 

230 It calls ``previsit()`` and ``postvisit()`` as follows: 

231 

232 * ``previsit(node, par_value)`` - should return ``(par_value, value)`` 

233 ``par_value`` is as returned from ``previsit()`` of the parent. 

234 

235 * ``postvisit(node, par_value, value)`` - should return ``value`` 

236 ``par_value`` is as returned from ``previsit()`` of the parent, and ``value`` is as 

237 returned from ``previsit()`` of this node itself. The return ``value`` is ignored except 

238 the one for the root node, which is returned from the overall ``visit_tree()`` call. 

239 

240 For the initial node, ``par_value`` is None. ``postvisit`` may be None. 

241 """ 

242 if not postvisit: 

243 postvisit = lambda node, pvalue, value: None 

244 

245 iter_children = iter_children_func(node) 

246 done = set() 

247 ret = None 

248 stack = [(node, None, _PREVISIT)] # type: List[Tuple[AstNode, Optional[Token], Union[Optional[Token], object]]] 

249 while stack: 

250 current, par_value, value = stack.pop() 

251 if value is _PREVISIT: 

252 assert current not in done # protect againt infinite loop in case of a bad tree. 

253 done.add(current) 

254 

255 pvalue, post_value = previsit(current, par_value) 

256 stack.append((current, par_value, post_value)) 

257 

258 # Insert all children in reverse order (so that first child ends up on top of the stack). 

259 ins = len(stack) 

260 for n in iter_children(current): 

261 stack.insert(ins, (n, pvalue, _PREVISIT)) 

262 else: 

263 ret = postvisit(current, par_value, cast(Optional[Token], value)) 

264 return ret

268def walk(node): 

269 # type: (AST) -> Iterator[Union[Module, AstNode]] 

270 """ 

271 Recursively yield all descendant nodes in the tree starting at ``node`` (including ``node`` 

272 itself), using depth-first pre-order traversal (yieling parents before their children). 

273 

274 This is similar to ``ast.walk()``, but with a different order, and it works for both ``ast`` and 

275 ``astroid`` trees. Also, as ``iter_children()``, it skips singleton nodes generated by ``ast``. 

276 """ 

277 iter_children = iter_children_func(node) 

278 done = set() 

279 stack = [node] 

280 while stack: 

281 current = stack.pop() 

282 assert current not in done # protect againt infinite loop in case of a bad tree. 

283 done.add(current) 

284 

285 yield current 

286 

287 # Insert all children in reverse order (so that first child ends up on top of the stack). 

288 # This is faster than building a list and reversing it. 

289 ins = len(stack) 

290 for c in iter_children(current): 

291 stack.insert(ins, c)

294def replace(text, replacements): 

295 # type: (str, List[Tuple[int, int, str]]) -> str 

296 """ 

297 Replaces multiple slices of text with new values. This is a convenience method for making code 

298 modifications of ranges e.g. as identified by ``ASTTokens.get_text_range(node)``. Replacements is 

299 an iterable of ``(start, end, new_text)`` tuples. 

300 

301 For example, ``replace("this is a test", [(0, 4, "X"), (8, 9, "THE")])`` produces 

302 ``"X is THE test"``. 

303 """ 

304 p = 0 

305 parts = [] 

306 for (start, end, new_text) in sorted(replacements): 

307 parts.append(text[p:start]) 

308 parts.append(new_text) 

309 p = end 

310 parts.append(text[p:]) 

311 return ''.join(parts)

402 def fstring_positions_work(): 

403 # type: () -> bool 

404 """ 

405 The positions attached to nodes inside f-string FormattedValues have some bugs 

406 that were fixed in Python 3.9.7 in https://github.com/python/cpython/pull/27729. 

407 This checks for those bugs more concretely without relying on the Python version. 

408 Specifically this checks: 

409 - Values with a format spec or conversion 

410 - Repeated (i.e. identical-looking) expressions 

411 - Multiline f-strings implicitly concatenated. 

412 """ 

413 source = """( 

414 f"a {b}{b} c {d!r} e {f:g} h {i:{j}} k {l:{m:n}}" 

415 f"a {b}{b} c {d!r} e {f:g} h {i:{j}} k {l:{m:n}}" 

416 f"{x + y + z} {x} {y} {z} {z} {z!a} {z:z}" 

417 )""" 

418 tree = ast.parse(source) 

419 name_nodes = [node for node in ast.walk(tree) if isinstance(node, ast.Name)] 

420 name_positions = [(node.lineno, node.col_offset) for node in name_nodes] 

421 positions_are_unique = len(set(name_positions)) == len(name_positions) 

422 correct_source_segments = all( 

423 ast.get_source_segment(source, node) == node.id 

424 for node in name_nodes 

425 ) 

426 return positions_are_unique and correct_source_segments
